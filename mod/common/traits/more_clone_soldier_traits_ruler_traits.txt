	is_councilor_trait_for_ruler = yes
	inline_script = {

		RARITY = common
		COUNCIL = triggered
		TIER = none
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = governor
		}

		army_defense_health_mult = 0.2
		army_defense_morale_mult = 0.2

	triggered_modifier = {
		potential = {
			is_ruler = no

		}
		science_ship_survey_speed = 0.25
		ship_sensor_range_add = 1
		shipclass_science_ship_disengage_mult = 0.25
		category_military_theory_research_speed_mult = 0.05 # just for flavor
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = admiral
		}
		ship_fire_rate_mult = 0.25
		ships_upkeep_mult = -0.1
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = general
		}
		army_morale = 0.1
		army_disengage_chance_mult = 0.2
		army_experience_gain_mult = 0.2
	}
	triggered_modifier = {
		potential = { is_ruler = yes }
		army_morale = 0.1
		country_ship_upgrade_cost_mult = -0.1
		country_naval_cap_mult = 0.1
		ship_speed_mult = 0.1
	}
	triggered_desc = {
		exclusive_trigger = {
			is_ruler = no
			leader_class = governor
		}
		text = leader_trait_clone_army_governor_desc
	}
	triggered_desc = {
		exclusive_trigger = {
			is_ruler = no
			leader_class = scientist
		}
		text = leader_trait_clone_army_scientist_desc
	}
	triggered_desc = {
		exclusive_trigger = {
			is_ruler = no
			leader_class = admiral
		}
		text = leader_trait_clone_army_admiral_desc
	}
	triggered_desc = {
		exclusive_trigger = {
			is_ruler = no
			leader_class = general
		}
		text = leader_trait_clone_army_general_desc
	}
	triggered_desc = {
		exclusive_trigger = { is_ruler = yes }
		text = leader_trait_clone_army_ruler_desc
	}
	leader_class = { admiral governor scientist general }
